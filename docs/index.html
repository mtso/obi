Obi is an experimental programming language.

Noteworthy features:

  • Trailing lambdas
  • Match expressions for control flow
  • Dynamic associative arrays

Examples

Dragon Turns

fun dragon(n) {
    fun dragon'(n, turn) {
        match n <= 0 {
            true -> return turn;
            _ -> ();
        };
        dragon'(n - 1, "R") + turn + dragon'(n - 1, "L");
    }

    return dragon'(n, "");
}

print(dragon(5));
// expect: "RRLRRLLRRRLLRLLRRRLRRLLLRRLLRLLRRLRRLLRRRLLRLLLRRLRRLLLRRLLRLL"

Adder

fun add(x) {
    return fun(y) {
        return x + y;
    };
}

print(add(1)(2)); // expect: "3"

List/Each

fun List() {
    self := [
        items = [],
        len = 0,
        add = fun(item) {
            self.items.(self.len) = item;
            self.len = self.len + 1;
        },
        get = fun(i) {
            return self.items.(i);
        },
        size = fun() {
            return self.len;
        },
    ];
    return self;
}

fun while(p, f) {
    match (p()) {
        false -> return;
        _ -> ();
    };
    f();
    while(p, f);
}

fun each(list, fn) {
    i := 0; while() { i < list.len; } {
        fn(list.get(i));
        i = i + 1;
    };
}

nums := List()
nums.add(1);
nums.add(2);
nums.add(3);
each(nums) fun(n) {
    print(n);
};
// expect: "1"
// expect: "2"
// expect: "3"

DSL

fun Markup() {
    _buf := "";
    return [
        head = fun(children) {
            _buf = _buf + "";
            children();
            _buf = _buf + "";
        },
        body = fun(children) {
            _buf = _buf + "";
            children();
            _buf = _buf + "";
        },
        text = fun(string) {
            _buf = _buf + "" + string + "";
        },
        render = fun() {
            return _buf;
        },
    ];
}

m := Markup();
head := m.head;
body := m.body;
text := m.text;

head() {
    body() {
        text("hi");
        text("bye");
    };
};

print(m.render());
// expect: "<head><body><text>hi</text><text>bye</text></body></head>"
