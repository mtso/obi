fun greet(name) {
    print("Hello, " + name + "!");
}

greet("Obi");
// expect: "Hello, Obi!"

When a function is the last parameter of a call, the function can be defined outside of the parentheses.

fun give3(f) {
    f(1);
    f(2);
    f(3);
}

give3() fun(n) {
    print(n);
};
// expect: "1"
// expect: "2"
// expect: "3"
// desugars to: give3(fun(n) { print(n); });

When a function does not have parameters, the fun keyword can be omitted.

fun twice(f) {
    f();
    f();
}

twice() {
    print("again");
};
// expect: "again"
// expect: "again"
// desugars to: twice(fun() { print("again"); });

Tail call optimization is implemented for functions where the last expression is a call to the defined function

fun fib(n) {
    fun fib'(n, a, b) {
        match n {
            1 -> n;
            0 -> n;
            _ -> ();
        }
        fib'(n - 1, b, a + b);
    }

    fib'(n, n - 1, 0);
}

twice() {
    print("again");
};
// expect: "again"
// expect: "again"
// desugars to: twice(fun() { print("again"); });
